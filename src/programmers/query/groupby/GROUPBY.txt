1. 고양이와 개는 몇 마리 있을까
SELECT ANIMAL_TYPE, COUNT(ANIMAL_TYPE) AS 'count' 
FROM ANIMAL_INS
GROUP BY ANIMAL_TYPE
ORDER BY ANIMAL_TYPE ASC;

2. 동명 동물 수 찾기
SELECT 
    NAME, 
    COUNT(NAME) 
FROM 
    ANIMAL_INS
GROUP BY 
    NAME 
HAVING 
    COUNT(NAME) > 1
ORDER BY 
    NAME ASC;

3. 입양 시각 구하기(1)
SELECT 
    HOUR(DATETIME) AS 'HOUR', 
    COUNT(*) AS 'COUNT'
FROM 
    ANIMAL_OUTS
WHERE 
    HOUR(DATETIME) BETWEEN 9 AND 19
GROUP BY 
    HOUR(DATETIME)
ORDER BY 
    HOUR(DATETIME) ASC;

4. 입양 시각 구하기(2)
SELECT 
    HOUR(DATETIME) AS 'HOUR',
    COUNT(*) AS 'COUNT'
FROM 
    ANIMAL_OUTS
GROUP BY 
    HOUR(DATETIME)
ORDER BY 
    HOUR(DATETIME) ASC;

SET @HOUR = -1;

SELECT
    (@HOUR := @HOUR + 1) AS HOUR,
    (SELECT
        COUNT(HOUR(DATETIME))
     FROM 
        ANIMAL_OUTS
     WHERE
        HOUR(DATETIME)=@HOUR) AS COUNT
FROM 
    ANIMAL_OUTS
WHERE
    @HOUR < 23;
